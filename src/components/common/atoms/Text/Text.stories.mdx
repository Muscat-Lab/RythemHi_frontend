# Label

서비스에서 공통으로 사용되는 텍스트컴포넌트 입니다.

# Props

```ts
interface InputProps {
  // 텍스트의 내용
  children: React.ReactChild;

  //텍스트의 크기!
  //'extraSmall', 'small', 'medium', 'large', 'extraLarge' 5가지 사이즈가 있다
  textSize: SizeType;

  // 텍스트의 굵기! 'light', 'normal', 'bold'의 3가지 사이즈가 있다.
  textType?: BoldType;

  // 추가 스타일 클래스가 있을때 사용!
  className?: string;
}
```
