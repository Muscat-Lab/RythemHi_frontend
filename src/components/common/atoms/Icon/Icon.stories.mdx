# Label

프로젝트 내에서 사용되어지는 아이콘을 표현합니다.

## Props

```ts
type IconSize = 'small' | 'medium' | 'large';

type SVGIconKeys =
  | 'homeIcon'
  | 'searchIcon'
  | 'backButtonIcon'; // 아이콘이 추가 될때마다 여기에 키 추가

interface IconProps {
  /**
   * Input과 연결되어지는 id 값.
   */
  iconName: SVGIconKeys;

  /**
   * Icon의 색상 값.
   */
  color: string;
  /**
   * Icon의 크기. (width, height 동일로 들어가는 값) small | medium | large
   */
  iconSize: IconSize;
  /**
   * custom class 하고 싶은 경우 props로 받아 적용.
   */
  className?: string;
}
```
