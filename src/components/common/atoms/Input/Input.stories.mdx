# Label

회원가입과 로그인에서 사용되는 인풋 컴포넌트 입니다.

# Props

```ts
interface InputProps {
  // input 타입
  type: string;

  // input 이름
  name: string;

  // input placeholder
  placeholder?: string;

  // input안에 들어가는 값
  value: string | number;

  // 최소 길이 제한
  minLength?: string;

  // 길이 (현재 password의 폰트가 달라서 기준 사이즈 정해주어야 일치함!)
  width: string;

  // 입력값을 받는 함수
  onChange: (
    // eslint-disable-next-line no-unused-vars
    e: React.ChangeEvent<HTMLInputElement>,
  ) => void;
}
```
