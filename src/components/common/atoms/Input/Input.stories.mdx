# Input

회원가입과 로그인에서 사용되는 인풋 컴포넌트 입니다.

# Props

```ts
interface InputProps {
  // input의 사이즈 단위! 'small', 'medium','large'가 있다.
  inputSize: SizeType;
  // 입력 성공 or 실패에 따라 border-color를 다르게 주기! 'success', 'error'가 있다.
  variant?: VariantType;
  // 사용자가 입력한 경우 배경색을 다르게 해주어야 하기 때문에 입력된 길이를 바탕으로
  //'active'와 'none'이 있다.
  active?: ActiveType;
  // Custom class
  className?: string;
}
```
